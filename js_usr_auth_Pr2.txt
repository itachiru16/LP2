  <script>
    // Hardcoded user credentials
    const validUsername = "user123";
    const validPassword = "pass123";

    document.getElementById("Form").addEventListener("submit", function(event) {
      event.preventDefault(); // Prevent form submission

      let a = document.getElementById("user").value;
      let b = document.getElementById("pass").value;

      if (a === validUsername && b === validPassword) {
        alert("Valid Username Or Pass");
        document.getElementById("msg").innerHTML="Login Successful";
      } else {
        alert("InValid Username Or Pass");
        document.getElementById("msg").innerHTML="Login Unsuccessful";
      }
    });
